The trend intelligence system enriches code analysis with real-time data from Google Search, GitHub, and Hacker News. It uses an agentic pipeline with RAG caching to provide context about technology momentum, risks, and opportunities.

## Pipeline architecture

The `TrendPipeline` orchestrates a 7-step agentic workflow:

<Steps>
  <Step title="RAG cache check">
    Query Supabase pgvector for recent insights (7-day window)
  </Step>
  
  <Step title="Query planning">
    Generate sub-queries optimized for each data source
  </Step>
  
  <Step title="Parallel multi-source search">
    Execute searches across Serper, GitHub, and Hacker News concurrently
  </Step>
  
  <Step title="Signal extraction">
    Parse results to extract version numbers, momentum indicators, and risk signals
  </Step>
  
  <Step title="Ranking and deduplication">
    Score results by relevance and remove duplicates
  </Step>
  
  <Step title="LLM synthesis">
    Use GPT-4 to synthesize findings into actionable insights
  </Step>
  
  <Step title="RAG cache storage">
    Store results in pgvector for future queries
  </Step>
</Steps>

async def search_all(
    serper_queries: list[str],
    github_queries: list[str],
    hn_queries: list[str],
    timeout: float = 30.0,
) -> list[SearchResult]:

    async with httpx.AsyncClient(timeout=timeout) as client:

            _search_github_batch(client, github_queries),
            _search_hn_batch(client, hn_queries),
        ]
        results = await asyncio.gather(
            *tasks,
            return_exceptions=True,
        )
    
    # Merge and return all results
    all_results = []
    for result in results:
        if not isinstance(result, Exception):
            all_results.extend(result)
    
    return all_results

async def _search_serper_single(
    client: httpx.AsyncClient,
    query: str,
    api_key: str,
) -> list[SearchResult]:
    """Execute a single Serper search query."""
    resp = await client.post(
        "https://google.serper.dev/search",
        headers={
            "X-API-KEY": api_key,
            "Content-Type": "application/json",
        },
        json={"q": query, "num": 5},
    )
    
    data = resp.json()
    results = []
    
    for item in data.get("organic", [])[:5]:
        results.append(SearchResult(
            title=item.get("title", ""),
            snippet=item.get("snippet", ""),
            url=item.get("link", ""),
            source=SourceType.SERPER,
            score=item.get("position", 0),
        ))
    
    return results
```

async def _search_github_repos(
    client: httpx.AsyncClient,
    query: str,
) -> list[SearchResult]:
    """Search GitHub repositories."""
    headers = {"Accept": "application/vnd.github.v3+json"}
    github_token = os.getenv("GITHUB_TOKEN")
    if github_token:
        headers["Authorization"] = f"Bearer {github_token}"
    
    resp = await client.get(
        "https://api.github.com/search/repositories",
        params={
            "q": query,
            "sort": "stars",
            "order": "desc",
            "per_page": 5,
        },
        headers=headers,
    )
    
    items = resp.json().get("items", [])
    results = []
    
    for repo in items:
        results.append(SearchResult(
            title=repo.get("full_name", ""),
            snippet=repo.get("description", ""),
            url=repo.get("html_url", ""),
            source=SourceType.GITHUB_SEARCH,
            score=repo.get("stargazers_count", 0),
            raw_data={
                "stars": repo.get("stargazers_count", 0),
                "forks": repo.get("forks_count", 0),
                "language": repo.get("language", ""),
                "topics": repo.get("topics", [])[:5],
            },
        ))
    
    return results

async def _search_hn_single(
    client: httpx.AsyncClient,
    query: str,
) -> list[SearchResult]:
    """Search HN via Algolia API."""
    resp = await client.get(
        "https://hn.algolia.com/api/v1/search",
        params={
            "query": query,
            "tags": "story",
            "hitsPerPage": 10,
        },
    )
    
    hits = resp.json().get("hits", [])
    results = []
    
    for hit in hits[:5]:
        hn_url = (
            hit.get("url", "")
            or f"https://news.ycombinator.com/item?id={hit.get('objectID', '')}"
        )
        results.append(SearchResult(
            title=hit.get("title", ""),
            snippet="",
            url=hn_url,
            source=SourceType.HACKER_NEWS,
            score=hit.get("points", 0) or 0,
            published_at=hit.get("created_at", "")[:10],
            raw_data={
                "objectID": hit.get("objectID", ""),
                "num_comments": hit.get("num_comments", 0),
                "author": hit.get("author", ""),
            },
        ))
    
    return results
```

```python
def rank_results(
    results: list[SearchResult],
    signals: dict,
    tag: str,
) -> list[SearchResult]:
    """Score and sort results by relevance."""
    for result in results:
        score = 0
        
        # Source weight
        if result.source == SourceType.GITHUB_SEARCH:
            score += result.score * 0.1  # GitHub stars
        elif result.source == SourceType.HACKER_NEWS:
            score += result.score * 2  # HN points
        
        # Recency boost
        if result.published_at:
            days_old = (datetime.now() - parse_date(result.published_at)).days
            if days_old < 30:
                score += 10
        
        # Tag relevance
        if tag.lower() in result.title.lower():
            score += 5
        
        result.score = score
    
    return sorted(results, key=lambda r: r.score, reverse=True)
```

- Opportunities (3 max)

{context}

    return parse_insight(response["content"], tag, ranked_results)

From backend/src/advisor/trends/pipeline.py:148-183:

```python
async def run_batch(
    self,
    tags: list[str],
    force_refresh: bool = True,
    max_concurrent: int = 1,
) -> dict[str, TrendInsight]:
    """Run pipeline for multiple tags in parallel.
    
    Uses a semaphore to cap concurrency and avoid API rate limits.
    """
    sem = asyncio.Semaphore(max_concurrent)
    
    async def _analyze_one(tag: str) -> tuple[str, TrendInsight | None]:
        async with sem:
            try:
                insight = await self.analyze_tag(
                    tag,
                    force_refresh=force_refresh,
                )
                return (tag, insight)
            except Exception as e:
                logger.error(f"Pipeline failed for '{tag}': {e}")
                return (tag, None)
    
    pairs = await asyncio.gather(
        *[_analyze_one(t) for t in tags],
    )
    return {tag: ins for tag, ins in pairs if ins is not None}
```
